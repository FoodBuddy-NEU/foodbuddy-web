在浏览器Console运行此脚本:

// 检查一个灰色文本元素的实际CSS
const grayText = document.querySelector('.text-gray-600, .text-gray-500, .text-neutral-600');
if (grayText) {
  console.log('=== 元素类名 ===');
  console.log(grayText.className);
  
  console.log('\n=== 计算后的颜色 ===');
  console.log('Color:', window.getComputedStyle(grayText).color);
  
  console.log('\n=== 所有应用的CSS规则 ===');
  const styles = window.getComputedStyle(grayText);
  console.log('color:', styles.color);
  console.log('font-size:', styles.fontSize);
}

// 检查所有带有 dark:text-white 的元素
console.log('\n=== 所有 dark:text-white 元素 ===');
document.querySelectorAll('[class*="dark:text-white"]').forEach((el, i) => {
  const computed = window.getComputedStyle(el).color;
  console.log(`${i + 1}. ${el.className.substring(0, 50)}... → ${computed}`);
});

// 检查HTML是否有dark类
console.log('\n=== Dark模式状态 ===');
console.log('Has dark class:', document.documentElement.classList.contains('dark'));
console.log('HTML classes:', document.documentElement.className);

或者运行这个来强制检查所有文本颜色:

// 找出所有不是白色的文本
document.querySelectorAll('*').forEach(el => {
  const color = window.getComputedStyle(el).color;
  const text = el.textContent?.trim();
  // 检查是否是灰色 (rgb值不全是255或0)
  if (color.includes('rgb') && !color.includes('255, 255, 255') && text && text.length < 50) {
    console.log(`Text: "${text.substring(0, 30)}" | Color: ${color} | Class: ${el.className}`);
  }
});
